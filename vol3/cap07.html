Funções em Python

# Função simples def saudacao(nome): return "Olá, " + nome + "!" mensagem = saudacao("Fernando") print(mensagem) # Olá, Fernando! # Função com múltiplos parâmetros def soma(a, b): return a + b print(soma(10, 25)) # 35 # Função com parâmetro padrão def apresentar(nome, idade=18): print(f"Nome: {nome}, Idade: {idade}") apresentar("Carlos") # Nome: Carlos, Idade: 18 apresentar("Maria", 25) # Nome: Maria, Idade: 25 # Recursividade — Fatorial def fatorial(n): if n <= 1: return 1 else: return n * fatorial(n - 1) print(fatorial(5)) # 120
